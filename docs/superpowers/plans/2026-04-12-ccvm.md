# ccvm Implementation Plan

> **For agentic workers:** REQUIRED SUB-SKILL: Use superpowers:subagent-driven-development (recommended) or superpowers:executing-plans to implement this plan task-by-task. Steps use checkbox (`- [ ]`) syntax for tracking.

**Goal:** Lima VM 内で Claude Code を auto mode で実行し、Docker も使える隔離開発環境を `ccvm` コマンドで管理する。

**Architecture:** Lima YAML で VM を定義し、`ccvm` シェルスクリプトで VM のライフサイクル（create/start/shell/stop/delete）を管理する。ワークスペースは RW マウント、Claude Code 設定と認証情報は RO マウント。

**Tech Stack:** Lima, bash, Docker Engine, Claude Code CLI

```yaml
# ccvm - Claude Code VM
# Lima VM definition for running Claude Code in auto mode with Docker support.
#
# Usage: This file is referenced by the ccvm script.
# Do not start directly with limactl — use `ccvm` instead.

images:
  - location: "https://cloud-images.ubuntu.com/releases/24.04/release/ubuntu-24.04-server-cloudimg-amd64.img"
    arch: "x86_64"
  - location: "https://cloud-images.ubuntu.com/releases/24.04/release/ubuntu-24.04-server-cloudimg-arm64.img"
    arch: "aarch64"

cpus: 1
memory: "2GiB"
disk: "10GiB"

# Mounts are configured dynamically by the ccvm script.
# The workspace path is passed as an argument and mounted at /workspace (RW).
# ~/.claude/ and ~/.config/gh/ are mounted read-only for settings and auth.
mounts: []

provision:
  - mode: system
    script: |
      #!/bin/bash
      set -eux

      # Docker Engine (official repository)
      apt-get update
      apt-get install -y ca-certificates curl
      install -m 0755 -d /etc/apt/keyrings
      curl -fsSL https://download.docker.com/linux/ubuntu/gpg -o /etc/apt/keyrings/docker.asc
      chmod a+r /etc/apt/keyrings/docker.asc
      echo "deb [arch=$(dpkg --print-architecture) signed-by=/etc/apt/keyrings/docker.asc] https://download.docker.com/linux/ubuntu $(. /etc/os-release && echo "$VERSION_CODENAME") stable" > /etc/apt/sources.list.d/docker.list
      apt-get update
      apt-get install -y docker-ce docker-ce-cli containerd.io docker-buildx-plugin docker-compose-plugin

      # Add default user to docker group
      usermod -aG docker "${LIMA_CIDATA_USER}"

      # Basic tools
      apt-get install -y git ripgrep fd-find jq fzf

      # GitHub CLI
      curl -fsSL https://cli.github.com/packages/githubcli-archive-keyring.gpg | dd of=/usr/share/keyrings/githubcli-archive-keyring.gpg
      chmod go+r /usr/share/keyrings/githubcli-archive-keyring.gpg
      echo "deb [arch=$(dpkg --print-architecture) signed-by=/usr/share/keyrings/githubcli-archive-keyring.gpg] https://cli.github.com/packages stable main" > /etc/apt/sources.list.d/github-cli.list
      apt-get update
      apt-get install -y gh

      # Clean up
      apt-get clean
      rm -rf /var/lib/apt/lists/*

  - mode: user
    script: |
      #!/bin/bash
      set -eux

      # Node.js via nvm
      curl -o- https://raw.githubusercontent.com/nvm-sh/nvm/v0.40.3/install.sh | bash
      export NVM_DIR="$HOME/.nvm"
      [ -s "$NVM_DIR/nvm.sh" ] && . "$NVM_DIR/nvm.sh"
      nvm install --lts

      # Claude Code
      npm install -g @anthropic-ai/claude-code
```

### Task 3: ccvm スクリプトを作成

**Files:**
- Create: `bin/ccvm`

- [ ] **Step 1: bin/ccvm を作成**

```bash
#!/bin/bash
set -euo pipefail

CCVM_YAML="$(cd "$(dirname "$(readlink -f "$0")")/../config/lima" && pwd)/ccvm.yaml"
CCVM_PREFIX="ccvm"

usage() {
    cat <<EOF
Usage:
  ccvm [<workspace-path>]    Start Claude Code in a Lima VM
  ccvm list                  List ccvm VMs
  ccvm stop <name> [--all]   Stop a ccvm VM (or all with --all)
  ccvm delete <name> [--all] Delete a ccvm VM (or all with --all)
EOF
}

# Resolve workspace path and derive VM name
resolve_workspace() {
    local ws="${1:-.}"
    ws="$(cd "$ws" && pwd)"
    echo "$ws"
}

vm_name_from_path() {
    local ws="$1"
    echo "${CCVM_PREFIX}-$(basename "$ws")"
}

# Check if a VM exists
vm_exists() {
    limactl list --json 2>/dev/null | jq -e --arg name "$1" '.[] | select(.name == $name)' >/dev/null 2>&1
}

# Get VM status (Running, Stopped, etc.)
vm_status() {
    limactl list --json 2>/dev/null | jq -r --arg name "$1" '.[] | select(.name == $name) | .status'
}

# List all ccvm VMs
cmd_list() {
    limactl list --json 2>/dev/null | jq -r --arg prefix "${CCVM_PREFIX}-" '.[] | select(.name | startswith($prefix)) | "\(.name)\t\(.status)\t\(.cpus)cpu\t\(.memory)"'
}

# Stop VM(s)
cmd_stop() {
    if [ "${1:-}" = "--all" ]; then
        local vms
        vms=$(limactl list --json 2>/dev/null | jq -r --arg prefix "${CCVM_PREFIX}-" '.[] | select(.name | startswith($prefix)) | select(.status == "Running") | .name')
        for vm in $vms; do
            echo "Stopping $vm..."
            limactl stop "$vm"
        done
        return
    fi

    local name="${1:?VM name required. Usage: ccvm stop <name>}"
    local full_name="${CCVM_PREFIX}-${name}"
    limactl stop "$full_name"
}

# Delete VM(s)
cmd_delete() {
    if [ "${1:-}" = "--all" ]; then
        local vms
        vms=$(limactl list --json 2>/dev/null | jq -r --arg prefix "${CCVM_PREFIX}-" '.[] | select(.name | startswith($prefix)) | .name')
        for vm in $vms; do
            echo "Deleting $vm..."
            limactl stop "$vm" 2>/dev/null || true
            limactl delete "$vm" --force
        done
        return
    fi

    local name="${1:?VM name required. Usage: ccvm delete <name>}"
    local full_name="${CCVM_PREFIX}-${name}"
    limactl stop "$full_name" 2>/dev/null || true
    limactl delete "$full_name" --force
}

# Start Claude Code in a VM
cmd_start() {
    local ws
    ws="$(resolve_workspace "${1:-.}")"
    local name
    name="$(vm_name_from_path "$ws")"

    local claude_dir="$HOME/.claude"
    local gh_dir="$HOME/.config/gh"

    if ! vm_exists "$name"; then
        echo "Creating VM $name..."
        limactl create \
            --name "$name" \
            --mount "type=reverse-sshfs,location=$ws,mountPoint=/workspace,writable=true" \
            --mount "type=reverse-sshfs,location=$claude_dir,mountPoint=/home/${USER}.linux/.claude,writable=false" \
            --mount "type=reverse-sshfs,location=$gh_dir,mountPoint=/home/${USER}.linux/.config/gh,writable=false" \
            "$CCVM_YAML"
        limactl start "$name"
    else
        local status
        status="$(vm_status "$name")"
        if [ "$status" != "Running" ]; then
            echo "Starting VM $name..."
            limactl start "$name"
        fi
    fi

    echo "Connecting to $name (workspace: $ws)..."
    limactl shell "$name" -- bash -lc "cd /workspace && claude --permission-mode auto"
}

# Main
case "${1:-}" in
    -h|--help|help)
        usage
        ;;
    list)
        cmd_list
        ;;
    stop)
        shift
        cmd_stop "${@}"
        ;;
    delete)
        shift
        cmd_delete "${@}"
        ;;
    *)
        cmd_start "${1:-}"
        ;;
esac
```
